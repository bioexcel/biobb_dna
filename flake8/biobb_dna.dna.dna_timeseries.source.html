20 class HelParTimeSeries(BiobbObject):
21 """
22 | biobb_dna HelParTimeSeries
23 | Created time series and histogram plots for each base pair from a helical parameter series file.
24 | The helical parameter series file is expected to be a table, with the first column being an index and the rest the helical parameter values for each base/basepair.
25  
26 Args:
27 input_ser_path (str): Path to .ser file for helical parameter. File is expected to be a table, with the first column being an index and the rest the helical parameter values for each base/basepair. File type: input. `Sample file <https://raw.githubusercontent.com/bioexcel/biobb_dna/master/biobb_dna/test/data/dna/canal_output_shift.ser>`_. Accepted formats: ser (edam:format_2330).
28 output_zip_path (str): Path to output .zip files where data is saved. File type: output. `Sample file <https://raw.githubusercontent.com/bioexcel/biobb_dna/master/biobb_dna/test/reference/dna/timeseries_output.zip>`_. Accepted formats: zip (edam:format_3987).
29 properties (dict):
30 * **sequence** (*str*) - (None) Nucleic acid sequence corresponding to the input .ser file. Length of sequence is expected to be the same as the total number of columns in the .ser file, minus the index column (even if later on a subset of columns is selected with the *usecols* option).
31 * **bins** (*int*) - (None) Bins for histogram. Parameter has same options as matplotlib.pyplot.hist.
32 * **helpar_name** (*str*) - (None) Helical parameter name. It must match the name of the helical parameter in the .ser input file. Values: majd, majw, mind, minw, inclin, tip, xdisp, ydisp, shear, stretch, stagger, buckle, propel, opening, rise, roll, twist, shift, slide, tilt, alphaC, alphaW, betaC, betaW, gammaC, gammaW, deltaC, deltaW, epsilC, epsilW, zetaC, zetaW, chiC, chiW, phaseC, phaseW.
33 * **stride** (*int*) - (1000) granularity of the number of snapshots for plotting time series.
34 * **seqpos** (*list*) - (None) list of sequence positions (columns indices starting by 1) to analyze. If not specified it will analyse the complete sequence.
35 * **remove_tmp** (*bool*) - (True) [WF property] Remove temporal files.
36 * **restart** (*bool*) - (False) [WF property] Do not execute if output files exist.
37 * **sandbox_path** (*str*) - ("./") [WF property] Parent path to the sandbox directory.
38  
39 Examples:
40 This is a use example of how to use the building block from Python::
41  
42 from biobb_dna.dna.dna_timeseries import dna_timeseries
43  
44 prop = {
45 'helpar_name': 'twist',
46 'seqpos': [1,2,3,4,5],
47 'sequence': 'GCAACGTGCTATGGAAGC',
48 }
49 dna_timeseries(
50 input_ser_path='/path/to/twist.ser',
51 output_zip_path='/path/to/output/file.zip'
52 properties=prop)
53 Info:
54 * wrapped_software:
55 * name: In house
56 * license: Apache-2.0
57 * ontology:
58 * name: EDAM
59 * schema: http://edamontology.org/EDAM.owl
60  
61 """

122 def launch(self) -> int:
123 """Execute the :class:`HelParTimeSeries <dna.dna_timeseries.HelParTimeSeries>` object."""
124  
125 # Setup Biobb
126 if self.check_restart():
127 return 0
128 self.stage_files()
129  
130 # check sequence
131 if self.sequence is None or len(self.sequence) < 2:
132 raise ValueError("sequence is null or too short!")
133  
134 # calculate cols with 0 index
135 if self.seqpos is not None:
136 cols = [i-1 for i in self.seqpos]
137 else:
138 cols = list(range(len(self.sequence)))
139  
140 # sort cols in ascending order
141 cols.sort()
142  
143 # check seqpos for base pairs
144 if self.seqpos is not None and self.helpar_name in constants.hp_basepairs:
145 if (max(cols) > len(self.sequence) - 2) or (min(cols) < 0):
146 raise ValueError(
147 f"seqpos values must be between 1 and {len(self.sequence) - 1}")
148 if not (isinstance(self.seqpos, list) and len(self.seqpos) > 1):
149 raise ValueError(
150 "seqpos must be a list of at least two integers")
151 # check seqpos for non base pairs
152 elif self.seqpos is not None and self.helpar_name not in constants.hp_basepairs:
153 if (max(cols) > len(self.sequence) - 1) or (min(cols) < 0):
154 raise ValueError(
155 f"seqpos values must be between 1 and {len(self.sequence)}")
156 if not (isinstance(self.seqpos, list) and len(self.seqpos) > 1):
157 raise ValueError(
158 "seqpos must be a list of at least two integers")
159  
160 if self.helpar_name in constants.hp_basepairs:
161 # remove first and last base pairs from cols if they match 0 and len(sequence)
162 if min(cols) == 0:
163 cols.pop(0)
164 if max(cols) == len(self.sequence) - 1:
165 cols.pop(-1)
166  
167 # discard first and last base(pairs) from sequence
168 sequence = self.sequence[1:-1]
169 # create indices list
170 indices = cols.copy()
171 # create subunits list from cols
172 subunits = [f"{i+1}_{sequence[i-1:i+self.baselen]}" for i in cols]
173 # clean subunits (leave only basepairs)
174 pattern = re.compile(r'\d+_[A-Za-z]{2}')
175 # get removed items
176 removed_items = [s for s in subunits if not pattern.fullmatch(s)]
177 # get indices of removed items (in integer format and starting from 0)
178 removed_numbers = [int(match.group()) for item in removed_items if (match := re.match(r'\d+', item))]
179 removed_numbers = list(map(int, removed_numbers))
180 removed_numbers = [int(i)-1 for i in removed_numbers]
181 # remove non basepairs from subunits and indices
182 subunits = [s for s in subunits if pattern.fullmatch(s)]
183 indices = [i for i in indices if i not in removed_numbers]
184 else:
185 sequence = self.sequence
186 # create indices list
187 indices = cols.copy()
188 # trick for getting the index column from the .ser file
189 indices.insert(0, 0)
190 # create subunits list from cols
191 subunits = [f"{i+1}_{sequence[i:i+1+self.baselen]}" for i in cols]
192  
193 # read input .ser file
194 ser_data = read_series(
195 self.stage_io_dict['in']['input_ser_path'],
196 usecols=indices)
197  
198 # get columns for selected bases
199 ser_data.columns = subunits
200  
201 # write output files for all selected bases (one per column)
202 zf = zipfile.ZipFile(
203 Path(self.stage_io_dict["out"]["output_zip_path"]), "w")
204 for col in ser_data.columns:
205 # unstack columns to prevent errors from repeated base pairs
206 column_data = (
207 ser_data[[col]]
208 .unstack()
209 .dropna()
210 .reset_index(drop=True))
211 column_data.name = col
212 fu.log(f"Computing base number {col}...")
213  
214 # column series
215 series_colfn = f"series_{self.helpar_name}_{col}"
216 column_data.to_csv(
217 Path(self.stage_io_dict.get("unique_dir", "")) / f"{series_colfn}.csv")
218 # save table
219 zf.write(
220 Path(self.stage_io_dict.get("unique_dir", "")) / f"{series_colfn}.csv", arcname=f"{series_colfn}.csv")
221  
222 fig, axs = plt.subplots(1, 1, dpi=300, tight_layout=True)
223 reduced_data = column_data.iloc[::self.stride]
224 axs.plot(reduced_data.index, reduced_data.to_numpy())
225 axs.set_xlabel("Time (Snapshots)")
226 axs.set_ylabel(f"{self.helpar_name.capitalize()} ({self.hp_unit})")
227 axs.set_title(
228 f"Helical Parameter vs Time: {self.helpar_name.capitalize()} "
229 "(base pair "
  • E225 Missing whitespace around operator
230 f"{'step' if self.baselen==1 else ''} {col})")
231 fig.savefig(
232 Path(self.stage_io_dict.get("unique_dir", "")) / f"{series_colfn}.jpg", format="jpg")
233 # save plot
234 zf.write(
235 Path(self.stage_io_dict.get("unique_dir", "")) / f"{series_colfn}.jpg", arcname=f"{series_colfn}.jpg")
236 plt.close()
237  
238 # columns histogram
239 hist_colfn = f"hist_{self.helpar_name}_{col}"
240 fig, axs = plt.subplots(1, 1, dpi=300, tight_layout=True)
241 ybins, x, _ = axs.hist(column_data, bins=self.bins)
242 pd.DataFrame({self.helpar_name: x[:-1], "density": ybins}).to_csv(
243 Path(self.stage_io_dict.get("unique_dir", "")) / f"{hist_colfn}.csv",
244 index=False)
245 # save table
246 zf.write(
247 Path(self.stage_io_dict.get("unique_dir", "")) / f"{hist_colfn}.csv",
248 arcname=f"{hist_colfn}.csv")
249  
250 axs.set_ylabel("Density")
251 axs.set_xlabel(f"{self.helpar_name.capitalize()} ({self.hp_unit})")
252 fig.savefig(
253 Path(self.stage_io_dict.get("unique_dir", "")) / f"{hist_colfn}.jpg",
254 format="jpg")
255 # save plot
256 zf.write(
257 Path(self.stage_io_dict.get("unique_dir", "")) / f"{hist_colfn}.jpg",
258 arcname=f"{hist_colfn}.jpg")
